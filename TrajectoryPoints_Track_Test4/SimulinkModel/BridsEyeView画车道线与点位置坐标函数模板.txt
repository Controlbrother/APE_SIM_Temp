function y = fcn(X,Y)

y = zeros(length(X),9);
for i = 1:1:length(X)
    y(i,1) = X(i);
    y(i,2) = Y(i);
    y(i,3) = 1;
    y(i,4) = 0;
    y(i,5) = 0;
    y(i,6) = i;
    y(i,7) = 1;
    y(i,8) = 1;
end
end

function y = fcn(C0,C1,C2,C3)

y = zeros(1,9);
y(1,1) = C3;
y(1,2) = C2;
y(1,3) = C1;
y(1,4) = C0;
y(1,5) = 0;
y(1,6) = 250;
y(1,7) = 0;
y(1,8) = 0;
y(1,9) = 1;
end
